\subsubsection{Notifications}
\begin{lstlisting}[language=sql]
/* function loadAllNotifications() { */
'SELECT   *
FROM     notifications
WHERE    user_id = ?
ORDER BY id DESC',
[\Auth::id()];

/* function loadUnreadNotifications() { */
'SELECT  *
FROM    notifications
WHERE   user_id = ?
AND     is_read = 0',
[\Auth::id()];

/* function loadLastNNotifications($n) { */
'SELECT   *
FROM     notifications
WHERE    user_id = ?
ORDER BY id DESC
LIMIT    ?',
[\Auth::id(), $n];

/* function updateNotificationsToSeen() { */
'UPDATE   notifications
SET      is_read = true
WHERE    user_id = ?',
[\Auth::id()];

/* function storeNotification($id, $type, $genId, $objectId = NULL) { */
'INSERT INTO notifications (user_id, generator_user_id, type, object_id)
VALUE (?, ?, ?, ?)',
[$id, $genId, $type, $objectId]);
\end{lstlisting}
